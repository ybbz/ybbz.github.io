获取案例库中所有标题并分词，统计词频并保存(Python实现)

发表于 2016-03-07   |   分类于 Data Mining   |  

首先从数据库（案例表case）中获取到所有案例的标题，并使用结巴中文分词组件进行批量中文分词。统计分词后的词频，并将结果保存到数据库（词频表corpus）。

# encoding=utf-8

import jieba
import mysql.connector
import datetime

# 连接数据库
conn = mysql.connector.connect(host="localhost", user="name", passwd="password", db="database")
cursor = conn.cursor()

# 查询案例库(案例名称)
cursor.execute('select name from case')
values = cursor.fetchall()
print("共查询到" + str(len(values)) + "条数据")

word_list = []  # 分词列表
word_dict = {}  # 分词词频字典
word_frequency = []  # 分词词频列表

file_origin = '/Users/ybbz/Desktop/case.txt'
file_count = '/Users/ybbz/Desktop/casecount.txt'


# 获得当前时间并格式化
def getTime():
    now = datetime.datetime.now()  # 时间数组格式
    format_time = now.strftime("%Y-%m-%d %H:%M:%S")  # 转换为指定格式
    return format_time


# 处理数据并保存文本
with open(file_origin, 'w') as f1:
    pass
with open(file_origin, 'a') as f1, open(file_count, 'w') as f2:
    count = 0
    for value in values:
        # 写入原句子
        f1.write(value[0] + "\n")
        # 原句子分词
        value_cut = jieba.cut_for_search(value[0])
        # 分词汇总
        for cut in value_cut:
            word_list.append(cut)
        # 计数器
        count += 1
    print("共处理" + str(count) + "条数据,进行了中文分词！")

    # 统计词频
    for word in word_list:
        if word not in word_dict:
            word_dict[word] = 1
        else:
            word_dict[word] += 1

    # 获取当前时间
    time = getTime()

    # 保存数据到文件
    for key in word_dict:
        f2.write(key + ' ' + str(word_dict[key]) + "\n")
        # 保存词频到列表
        item = (key, word_dict[key], time)
        word_frequency.append(item)
    print("共处理" + str(count) + "条数据,统计分词词频并保存到文件！")

    # 保存数据到数据库
    sql = "insert into corpus(name,frequency,create_time) values(%s, %s, %s)"
    cursor.executemany(sql, word_frequency)
    conn.commit()
    print("词频保存到数据库完成！")
